#!/bin/bash -l
#SBATCH --reservation uppmax2026-1-61_9
#SBATCH -A uppmax2026-1-61
#SBATCH -n 8
#SBATCH -t 48:00:00
#SBATCH -J braker3_Ricci
#SBATCH --mail-user=marco.ricci.evobio@gmail.com
#SBATCH --mail-type=ALL
#SBATCH -o /home/ricci/Genome_analysis_2026/5_gene_prediction/BRAKER3/logs/braker3_%j.out
#SBATCH -e /home/ricci/Genome_analysis_2026/5_gene_prediction/BRAKER3/logs/braker3_%j.err

set -euo pipefail

# -----------------------------
# MODULES
# -----------------------------
module load BRAKER/3.0.6-gfbf-2024a
module load AUGUSTUS/3.5.0-gfbf-2024a
module load SAMtools/1.22.1-GCC-13.3.0
module load Perl/5.38.2-GCCcore-13.3.0

# -----------------------------
# PATHS
# -----------------------------
ASSEMBLY="/home/ricci/Genome_analysis_2026/2_masking/1_polishing.fasta.masked"
BAM_DIR="/home/ricci/Genome_analysis_2026/5_gene_prediction/HiSat_RNA_mapping/bam"
OUT_DIR="/home/ricci/Genome_analysis_2026/5_gene_prediction/BRAKER3"
THREADS=8

mkdir -p ${OUT_DIR} ${OUT_DIR}/logs

# -----------------------------
# RACCOLTA AUTOMATICA DEI BAM
# -----------------------------
BAM_FILES=$(ls ${BAM_DIR}/*.bam | tr '\n' ',' | sed 's/,$//')

echo "### BAM files trovati:"
echo ${BAM_FILES}

# -----------------------------
# RUN BRAKER3
# -----------------------------
braker.pl \
    --genome=${ASSEMBLY} \
    --bam=${BAM_FILES} \
    --species=niphotrichum_augustus \
    --softmasking \
    --threads=${THREADS} \
    --workingdir=${OUT_DIR} \
    --gff3 \
    --etpmode

echo "### BRAKER3 completato"
